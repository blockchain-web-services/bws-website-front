```javascript
export function loadBWSProducts() {
  try {
    const docsData = fs.readFileSync(DOCS_INDEX_PATH, 'utf-8');
    const docsIndex = JSON.parse(docsData);

    const products = {};

    docsIndex.pages.forEach(page => {
      const productName = page.product;

      if (relevantProducts.includes(productName)) {
        if (!products[productName]) {
          products[productName] = {
            name: productName,
            url: page.url,
            category: page.category,
            description: page.summary || '',
            keywords: page.keywords || [],
            useCases: page.useCases || [],
            implementationSteps: page.implementationSteps || [],
            fullContent: page.fullContent || '',
            structuredContent: page.structuredContent || {},
            images: page.images || [] // ✅ ADD THIS LINE
          };
        } else {
          // Merge images from multiple pages for same product
          if (page.images && page.images.length > 0) {
            products[productName].images = [
              ...products[productName].images,
              ...page.images
            ];
          }
        }
      }
    });

    console.log(`✅ Loaded ${Object.keys(products).length} BWS products from documentation`);
    return products;
  } catch (error) {
    console.error(`❌ Error loading BWS products from docs: ${error.message}`);
    return {};
  }
}
```

### Phase 4: Twitter Image Upload Integration

**Goal**: Upload local image to Twitter and get media ID for tweet attachment

**File**: `scripts/crawling/production/reply-to-kol-posts.js`

**New Function**: Add before main workflow (around line 150)

```javascript
import fs from 'fs';
import path from 'path';

/**
 * Upload image to Twitter and return media ID
 *
 * @param {Object} client - Twitter API client (v1 for media upload)
 * @param {string} imagePath - Local path to image (e.g., "/assets/images/docs/x-bot/x-bot-0.png")
 * @returns {Promise<string>} Media ID for tweet attachment
 */
async function uploadImageToTwitter(client, imagePath) {
  try {
    // Convert relative path to absolute
    const absolutePath = path.join(process.cwd(), 'public', imagePath);

    if (!fs.existsSync(absolutePath)) {
      console.warn(`⚠️  Image not found: ${absolutePath}`);
      return null;
    }

    // Read image as buffer
    const imageBuffer = fs.readFileSync(absolutePath);

    // Upload to Twitter using v1 API (v2 doesn't support media upload yet)
    console.log(`📤 Uploading image: ${imagePath}`);
    const mediaId = await client.v1.uploadMedia(imageBuffer, {
      mimeType: 'image/png' // or detect from file extension
    });

    console.log(`✅ Image uploaded, media ID: ${mediaId}`);
    return mediaId;

  } catch (error) {
    console.error(`❌ Failed to upload image: ${error.message}`);
    return null; // Graceful degradation - post without image
  }
}
```

**Twitter API Notes**:
- Uses `client.v1.uploadMedia()` (v1 API)
- Returns media ID string (e.g., "1234567890")
- Max file size: 5MB for images
- Supported formats: PNG, JPG, GIF, WEBP

## Risk Mitigation

### Risk 1: Image Upload Failures
**Mitigation**: Graceful degradation
```javascript
if (imageUploadFailed) {
  console.warn('⚠️  Image upload failed, posting text-only reply');
  // Continue with text-only reply
}
```

### Risk 2: Missing/Invalid Images
**Mitigation**: File existence checks
```javascript
if (!fs.existsSync(absolutePath)) {
  console.warn(`⚠️  Image not found: ${absolutePath}`);
  return null; // Skip image attachment
}
```

### Risk 3: Twitter Rate Limits on Media Upload
**Mitigation**:
- Respect rate limits (track in `api-call-tracker.js`)
- Implement retry logic with exponential backoff
- Fall back to text-only if upload consistently fails

### Risk 4: Irrelevant/Low-Quality Images
**Mitigation**:
- Manual image curation in docs (docs team responsibility)
- Product owners can update docs images weekly
- First image is usually best (hero/overview shot)

### Risk 5: Image File Size Too Large
**Mitigation**:
```javascript
const MAX_IMAGE_SIZE = 5 * 1024 * 1024; // 5MB Twitter limit
const stats = fs.statSync(absolutePath);

if (stats.size > MAX_IMAGE_SIZE) {
  console.warn(`⚠️  Image too large: ${stats.size} bytes (max 5MB)`);
  return null;
}
```

## Decision Points

### Question 1: Image Attachment Rate
**Default**: 70% of replies include images
**Consideration**: Too many images might feel spammy, too few loses engagement benefit
**Recommendation**: Start at 50%, A/B test to find optimal rate

### Question 2: Image Selection Strategy
**Option A**: Always first image (simple, consistent)
**Option B**: Random selection from available images (variety)
**Option C**: Template-aware selection (complex, best results)
**Recommendation**: Start with Option A, evolve to Option C

### Question 3: Fallback Behavior
**When image upload fails or no images available**:
- ✅ Post text-only reply (recommended - graceful degradation)
- ❌ Skip reply entirely (too aggressive)
- ❌ Retry with delay (adds latency)

### Question 4: Image Dimensions
**Twitter Recommended**: 1200x675px (16:9 aspect ratio)
**Current Docs Images**: Various sizes (check after re-index)
**Action Needed**: Review docs images, potentially resize large images

### Question 5: Update Frequency
**Docs Indexer**: Weekly via GitHub Actions
**Images**: Update when docs change
**Recommendation**: Keep weekly cadence, manual trigger for urgent updates
